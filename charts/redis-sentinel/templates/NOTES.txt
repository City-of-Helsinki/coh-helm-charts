2. Connect to Redis:
   kubectl exec -it {{ include "redis-sentinel.fullname" . }}-redis-0 -n {{ .Values.global.namespace }} -- redis-cli -a $REDIS_PASSWORD

3. Connect to Sentinel:
   kubectl exec -it {{ include "redis-sentinel.fullname" . }}-sentinel-0 -n {{ .Values.global.namespace }} -- redis-cli -p {{ .Values.sentinel.config.port }} -a $REDIS_PASSWORD

4. Check Sentinel info:
   kubectl exec -it {{ include "redis-sentinel.fullname" . }}-sentinel-0 -n {{ .Values.global.namespace }} -- redis-cli -p {{ .Values.sentinel.config.port }} -a $REDIS_PASSWORD sentinel masters

5. Services:
   Redis: {{ include "redis-sentinel.fullname" . }}-headless:{{ .Values.redis.config.port }}
   Sentinel: {{ include "redis-sentinel.fullname" . }}-sentinel:{{ .Values.sentinel.config.port }}
